	socktype: INTEGER_32
			-- Denotes the type of socket that is wanted: SOCK_STREAM,
			-- SOCK_DGRAM, or SOCK_RAW.  When ai_socktype is zero the
			-- caller will accept any socket type

			-- Indicates which transport protocol is desired, IPPROTO_UDP
