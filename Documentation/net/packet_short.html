note
	description: "A data packet for sending and receiving on a socket."
	legal: "See notice at end of class."
	status: "See notice at end of class."
	date: "$Date: 2010-01-06 13:46:03 -0800 (Wed, 06 Jan 2010) $"
	revision: "$Revision: 82000 $"

class interface
	PACKET

create 
	make,
	make_from_managed_pointer

feature -- Initialization

	make (size: INTEGER_32)
			-- Create a packet.
		require
			valid_size: size >= 0

	make_from_managed_pointer (a_data: like data)
			-- Create packet from `a_data' memory stream.
		require
			a_data_not_void: a_data /= Void
		ensure
			data_set: data = a_data
	
feature -- Measurement

	count: INTEGER_32
			-- Number of elements in packet
	
feature -- Access

	element alias "[]" (i: INTEGER_32): CHARACTER_8 assign put_element
			-- Entry at index `i'.
			-- Was declared in PACKET as synonym of at.
		require
			valid_position: valid_position (i)

	at alias "@" (i: INTEGER_32): CHARACTER_8 assign put_element
			-- Entry at index `i'.
			-- Was declared in PACKET as synonym of element.
		require
			valid_position: valid_position (i)
	
feature -- Status report

	valid_position (i: INTEGER_32): BOOLEAN
			-- Is `i' within the bounds of Current?
	
feature -- Element change

	put_element (v: CHARACTER_8; i: INTEGER_32)
			-- Replace `i'-th entry by `v'.
		require
			valid_position: valid_position (i)
		ensure
			inserted: element (i) = v
	
feature -- Comparison

	is_equal (other: like Current): BOOLEAN
			-- Is current packet equal to `other'?
	
feature -- Duplication

	copy (other: like Current)
			-- Reinitialize by copying characters of `other'.
			-- (This is also used by clone)
		ensure then
			size_valid: count = other.count
	
feature -- Storage

	data: MANAGED_POINTER
			-- Place holder
	
invariant
	data_not_equal_void: data /= Void
