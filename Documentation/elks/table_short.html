	names: table, access
	access: key, membership
	contents: generic
	date: "$Date: 2012-05-23 21:13:10 -0700 (Wed, 23 May 2012) $"

deferred class interface
	TABLE [G, H]

feature -- Access

	item alias "[]" (k: H): G assign force
			-- Entry of key `k'.
			-- Was declared in TABLE as synonym of at.
		require
			valid_key: valid_key (k)

	at alias "@" (k: H): G assign force
			-- Entry of key `k'.
			-- Was declared in TABLE as synonym of item.
		require
			valid_key: valid_key (k)
	
feature -- Status report

	valid_key (k: H): BOOLEAN
			-- Is `k' a valid key?
	
feature -- Element change

	put (v: G; k: H)
			-- Associate value `v' with key `k'.
		require
			valid_key: valid_key (k)

	force (v: G; k: H)
			-- Associate value `v' with key `k'.
		require
			valid_key: valid_key (k)
		ensure
			inserted: item (k) = v
