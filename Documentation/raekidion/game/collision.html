note
	description: "[
		War of Raekidion - A collidable object
		A COLLISION is an entity capable of reacting to 
		being in the same space as another collidable object.
	]"
	author: "François Allard (binarmorker) and Marc-Antoine Renaud (Learz)"
	date: "$Date: May 23 2014$"
	revision: "$Revision: 1$"

deferred class 
	COLLISION

feature -- Access

	offset: INTEGER_32
		deferred
		end

	width: INTEGER_32
		deferred
		end

	height: INTEGER_32
		deferred
		end

	x: REAL_64
		deferred
		end

	y: REAL_64
		deferred
		end

	has_collided (a_other: COLLISION): BOOLEAN
			-- Has current's bounding area collided with another COLLISION
		local
			l_x_difference, l_y_difference: REAL_64
			l_center_distance_squared, l_radius_sum_squared: REAL_64
			l_first_offset_squared, l_second_offset_squared: REAL_64
		do
			l_x_difference := a_other.x - x
			l_y_difference := a_other.y - y
			l_center_distance_squared := (l_x_difference ^ 2.to_double) + (l_y_difference ^ 2.to_double)
			l_radius_sum_squared := ((a_other.width / 2) + (width / 2)) ^ 2.to_double
			if a_other.offset >= 0 then
				l_first_offset_squared := (a_other.offset ^ 2.to_double)
			else
				l_first_offset_squared := - (a_other.offset ^ 2.to_double)
			end
			if offset >= 0 then
				l_second_offset_squared := (offset ^ 2.to_double)
			else
				l_second_offset_squared := - (offset ^ 2.to_double)
			end
			Result := l_center_distance_squared - l_radius_sum_squared <= l_first_offset_squared - l_second_offset_squared
		end
