feature -- Fonctions -SDL.h

	frozen sdl_createwindow (a_title: POINTER; a_x, a_y, a_width, a_height: INTEGER_32; a_flags: NATURAL_32): POINTER
			-- Create a window

	frozen sdl_setwindowicon (a_window, a_surface: POINTER)
			-- Set the window icon

	frozen sdl_sethint (a_name: POINTER; a_value: POINTER)
			-- Set a hint with normal priority.

	frozen sdl_hintrenderscalequality: POINTER
			-- Hint that specifies scaling quality.

	frozen sdl_rendersetlogicalsize (a_renderer: POINTER; a_width, a_height: INTEGER_32)
			-- Set a device independent resolution for rendering.

	frozen sdl_destroywindow (a_window: POINTER)
			-- Destroy `a_window'

	frozen sdl_createrenderer (a_window: POINTER; a_index: INTEGER_32; a_flags: NATURAL_32): POINTER
			-- Create a renderer

	frozen sdl_destroyrenderer (a_renderer: POINTER)
			-- Destroy `a_renderer'

	frozen sdl_creatergbsurface (a_flags: NATURAL_32; a_width, a_height, a_depth: INTEGER_32; a_rmask, a_gmask, a_bmask, a_amask: NATURAL_32): POINTER
			-- Return `a_window' as a surface

	frozen sdl_settexturealphamod (a_texture: POINTER; a_value: NATURAL_8)

	frozen sdl_getwindowflags (a_window: POINTER): NATURAL_32
			-- Get the flags of `a_window'

	frozen sdl_quit
			-- Clean up all initialized subsystems

	frozen sdl_renderclear (a_renderer: POINTER)
			-- Clear `a_renderer' with the drawing color.

	frozen sdl_rendercopy (a_renderer, a_texture, a_srcrect, a_dstrect: POINTER)
			-- Copy a portion of `a_texture' to `a_renderer'.

	frozen sdl_rendercopyex (a_renderer, a_texture, a_srcrect, a_dstrect: POINTER; a_angle: REAL_64; a_center: POINTER; a_flip: NATURAL_32)
			-- Copy a portion `a_texture' to `a_renderer',
			-- Optionally rotate by `a_angle' around `a_center' and also flipping it top-bottom and/or left-right

	frozen sdl_flip_none: NATURAL_32
			-- Do not flip

	frozen sdl_renderpresent (a_renderer: POINTER)
			-- Update the screen with `a_renderer'

	frozen sdl_init (a_flags: NATURAL_32): NATURAL_32
			-- Initialize the SDL library
			-- Returns 0 on success or a negative error code on failure.

	frozen sdl_init_noreturn (a_flags: NATURAL_32)
			-- Initialize the SDL library
			-- Returns nothing.

	frozen sdl_savebmp (a_surface, a_filename: POINTER)
			-- Save `a_surface' as a BMP image in `a_filename'

	frozen sdl_loadimage (a_file: POINTER): POINTER
			-- Load `a_file' for use as an image in a new surface.

	frozen sdl_freesurface (a_surface: POINTER)
			-- Free `a_surface'

	frozen sdl_setrenderdrawcolor (a_renderer: POINTER; a_red, a_green, a_blue, a_alpha: NATURAL_8)
			-- Sets the color (`a_red', `a_green', `a_blue' and `a_alpha') in which to draw the next shape into `a_renderer'

	frozen sdl_getrenderdrawcolor (a_renderer, a_red, a_green, a_blue, a_alpha: POINTER)
			-- Gets the color (`a_red', `a_green', `a_blue' and `a_alpha') of `a_renderer'

	frozen sdl_renderfillrect (a_renderer, a_rect: POINTER)
			-- Draws `a_rect' inside `a_renderer'

	frozen sdl_createtexturefromsurface (a_renderer, a_surface: POINTER): POINTER
			-- Create a texture from `a_surface'.

	frozen sdl_renderreadpixels (a_renderer, a_rect: POINTER; a_format: NATURAL_32; a_pixels: POINTER; a_pitch: INTEGER_32)
			-- Read pixels from `a_rect' and write them into `a_pixels'.

	frozen sdl_destroytexture (a_texture: POINTER)
			-- Destroy `a_texture'

	frozen sdl_delay (a_time: INTEGER_32)
			-- Wait `a_time' milliseconds before returning.

	frozen sdl_getticks: NATURAL_32
			-- Number of milliseconds since the SDL library initialization.

	frozen set_sdl_rect_x (a_sdl_rect: POINTER; a_value: INTEGER_32)
			-- Move `a_sdl_rect' to `a_value' pixels on the horizontal axis.

	frozen set_sdl_rect_y (a_sdl_rect: POINTER; a_value: INTEGER_32)
			-- Move `a_sdl_rect' to `a_value' pixels on the vertical axis.

	frozen set_sdl_rect_w (a_sdl_rect: POINTER; a_value: INTEGER_32)
			-- Change the width of `a_sdl_rect' to `a_value' pixels.

	frozen set_sdl_rect_h (a_sdl_rect: POINTER; a_value: INTEGER_32)
			-- Change the height of `a_sdl_rect' to `a_value' pixels.

	frozen set_sdl_window_size (a_window: POINTER; a_w, a_h: INTEGER_32)
			-- Change the size of `a_window'

	frozen get_sdl_loadbmp_width (a_sdl_surface: POINTER): INTEGER_32
			-- Width of `a_sdl_surface'

	frozen get_sdl_loadbmp_height (a_sdl_surface: POINTER): INTEGER_32
			-- Height of `a_sdl_surface'

	frozen get_sdl_surface_pitch (a_sdl_surface: POINTER): INTEGER_32
			-- Number of pixels in a row on `a_sdl_surface'

	frozen get_sdl_surface_pixels (a_sdl_surface: POINTER): POINTER
			-- All pixels of `a_sdl_surface'

	frozen get_sdl_surface_format (a_sdl_surface: POINTER): POINTER
			-- Format of the pixels stored in `a_sdl_surface'

	frozen get_sdl_rect_x (a_sdl_rect: POINTER): INTEGER_32
			-- Horizontal position of `a_sdl_rect'

	frozen get_sdl_rect_y (a_sdl_rect: POINTER): INTEGER_32
			-- Vertical position of `a_sdl_rect'

	frozen get_sdl_rect_w (a_sdl_rect: POINTER): INTEGER_32
			-- Width of `a_sdl_rect'

	frozen get_sdl_rect_h (a_sdl_rect: POINTER): INTEGER_32
			-- Height of `a_sdl_rect'
