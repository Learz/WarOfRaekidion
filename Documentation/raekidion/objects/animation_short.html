note
	description: "[
		War of Raekidion - A simple animation
		An ANIMATION will switch between a fixed set of 
		different frames with time.
	]"
	author: "François Allard (binarmorker) and Marc-Antoine Renaud (Learz)"
	date: "$Date: May 23 2014$"
	revision: "$Revision: 1$"

class interface
	ANIMATION

create 
	make

feature -- Access

	lifetime: INTEGER_32
			-- Total time the animation has been running

	time: INTEGER_32
			-- Animation time

	frames: INTEGER_32
			-- Number of frames to animate

	current_frame: INTEGER_32
			-- The current frame to display

	update
			-- Update current frame
	
feature -- Status

	repeat: BOOLEAN
			-- If the animation must repeat endlessly

	passed_once: BOOLEAN
			-- If the animation played at least once
	
feature -- Element change

	set_x (a_x: REAL_64)
			-- Change x to `a_x'

	set_y (a_y: REAL_64)
			-- Change y to `a_y'
	
invariant
	current_frame_valid: current_frame < frames
	frames_not_null: frames > 0
	time_not_null: time > 0

	]"

end -- class ANIMATION
