note
	description: "[
		War of Raekidion - A scrolling background image
		A BACKGROUND is an image duplicated, then placed on top 
		of each other to create the impression of endless scrolling.
	]"
	author: "François Allard (binarmorker) and Marc-Antoine Renaud (Learz)"
	date: "$Date: May 23 2014$"
	revision: "$Revision: 1$"

class 
	BACKGROUND

inherit
	SPRITE
		rename
			make as sprite_make
		redefine
			update
		end

create 
	make

feature {NONE} -- Initialization

	make (a_name: STRING_8; a_window: WINDOW; a_x, a_y: INTEGER_16; a_scroll_speed: REAL_64)
			-- Initialize `Current' from `a_name', `a_window', `a_x', `a_y' and `a_scroll_speed'
		do
			origin_y := a_y
			scroll_speed := a_scroll_speed
			sprite_make (a_name, a_window, a_x.to_double, a_y.to_double)
			if scroll_speed > 0.to_double then
				wrapped_sprite := create {SPRITE}.make (a_name, a_window, a_x.to_double, a_y.to_integer_32 - height.to_double)
			elseif scroll_speed < 0.to_double then
				wrapped_sprite := create {SPRITE}.make (a_name, a_window, a_x.to_double, a_y.to_integer_32 + height.to_double)
			else
				wrapped_sprite := create {SPRITE}.make (a_name, a_window, a_x.to_double, a_y.to_double)
			end
		end
	
feature -- Access

	update
			-- Update `Current' movement
		do
			set_y (y + scroll_speed)
			wrapped_sprite.set_y (wrapped_sprite.y + scroll_speed)
			if y >= origin_y.to_integer_32 + height.to_double then
				set_y (origin_y.to_double)
				wrapped_sprite.set_y (origin_y.to_integer_32 - height.to_double)
			elseif y <= origin_y.to_integer_32 - height.to_double then
				set_y (origin_y.to_double)
				wrapped_sprite.set_y (origin_y.to_integer_32 + height.to_double)
			end
			wrapped_sprite.update
			Precursor {SPRITE}
		end
	
feature {NONE} -- Implementation

	scroll_speed: REAL_64
			-- Vertical scrolling speed (can be negative)

	origin_y: INTEGER_16
			-- Original y, for reference

	wrapped_sprite: SPRITE
			-- The superposed sprite

end -- class BACKGROUND
