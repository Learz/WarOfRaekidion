	description: "[
		An blob binding argument value for use with executing a SQLite statement.
	]"
	legal: "See notice at end of class."
	status: "See notice at end of class."
	date: "$Date: 2013-11-07 12:09:47 -0800 (Thu, 07 Nov 2013) $"
	revision: "$Revision: 93250 $"

class 
	SQLITE_BLOB_ARG

inherit
	SQLITE_BIND_ARG [MANAGED_POINTER]
		redefine
			is_valid_value
		end

create 
	make

feature -- Status report

	is_valid_value (a_value: like value): BOOLEAN
			-- Determines if an argument value is valid.
			--
			-- `a_value': The value to validate.
			-- `Result': True if the value is valid; False otherwise.
		do
			Result := attached a_value
		ensure then
			attached_a_value: Result implies attached a_value
		end
	
feature {SQLITE_STATEMENT} -- Basic operations

	bind_to_statement (a_statement: SQLITE_STATEMENT; a_index: INTEGER_32)
			-- Binds the argument to the statement, at a given index.
			--
			-- `a_statement': A prepared statement to bind the argument to.
			-- `a_index': The index to bind the value at.
		local
			l_value: like value
		do
			l_value := value
			check
				l_value_attached: attached l_value
			end
			sqlite_raise_on_failure ({SQLITE_EXTERNALS}.c_sqlite3_bind_blob (a_statement.internal_stmt, a_index, l_value.item, l_value.count, default_pointer))
		end
