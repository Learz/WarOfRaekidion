                        create l_arg_id.make (l_arg_variable)
                        l_arg_index := sqlite3_bind_parameter_index (l_api, l_stmt, l_arg_id.item)

                        end

            if sqlite_success (l_result) then

            end
        rescue

                l_locked := False
                l_db.unlock
