note
	description: "A network socket."
	legal: "See notice at end of class."
	status: "See notice at end of class."
	date: "$Date: 2011-10-24 16:52:05 -0700 (Mon, 24 Oct 2011) $"
	revision: "$Revision: 87603 $"

deferred class interface
	NETWORK_SOCKET

feature 

	exists: BOOLEAN
			-- Does socket exist?
		ensure then
			definition: Result implies descriptor_available

	close_socket

	descriptor: INTEGER_32
			-- Socket descriptor of current socket

	connect
			-- Connect socket to peer address.

	bind
			-- Bind socket to local address in address.
	
feature -- Status report

	address_type: NETWORK_SOCKET_ADDRESS
			-- Type of address and peer_address

	is_closed: BOOLEAN
			-- Is socket closed?

	is_created: BOOLEAN

	is_connected: BOOLEAN

	is_bound: BOOLEAN

	port: INTEGER_32
			-- Port socket is bound to.

	reuse_address: BOOLEAN
			-- Is reuse_address option set?
		require
			socket_exists: exists

	is_valid_peer_address (addr: attached like address): BOOLEAN
			-- Is `addr' a valid peer address?

	is_valid_family (addr: attached like address): BOOLEAN
			-- Is `addr' the same family as Current?

	ready_for_reading: BOOLEAN
			-- Is data available for reading from the socket within
			-- timeout seconds?
		require
			socket_exists: exists

	ready_for_writing: BOOLEAN
			-- Can data be written to the socket within timeout seconds?
		require
			socket_exists: exists

	has_exception_state: BOOLEAN
			-- Is socket in exception state within timeout seconds?
		require
			socket_exists: exists

	timeout: INTEGER_32
			-- Duration of timeout in seconds
	
feature -- Status setting

	set_reuse_address
			-- Turn reuse_address option on.
		require
			socket_exists: exists

	do_not_reuse_address
			-- Turn reuse_address option off.
		require
			socket_exists: exists

	set_timeout (n: INTEGER_32)
			-- Set timeout to `n' seconds.
		require
			non_negative: n >= 0
		ensure
			timeout_set: timeout = n or timeout = Default_timeout

	set_non_blocking
			-- Set socket to non-blocking mode.

	set_blocking
			-- Set socket to blocking mode.
	
invariant
	timeout_set: timeout > 0

end -- class NETWORK_SOCKET
