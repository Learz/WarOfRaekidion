			-- (from HELPER)

			l_key_bytes: SPECIAL [NATURAL_8]

feature {NONE} -- Implementation

	bit_count: NATURAL_64
			-- Bit count
			-- (from MESSAGE_DIGEST)
		do
			Result := byte_count |<< 3
		end

	byte_count: NATURAL_64
			-- Byte count
			-- (from MESSAGE_DIGEST)

	finish
			-- Finish updating, get ready to process the last block.
			-- (from MESSAGE_DIGEST)
		local
			length: NATURAL_64
		do
			length := bit_count
			message_pad
			process_length (length)
		end

	message_pad
			-- Pad with zero until 64bit is left in current block.
			-- (from MESSAGE_DIGEST)
		local
			pad_bytes: INTEGER_32
		do
			update_from_byte (128.to_natural_8)
			from
				pad_bytes := (56.to_natural_64 - (byte_count \\ 64.to_natural_64)).to_integer_32
				if pad_bytes < 0 then
					pad_bytes := pad_bytes + 64
				end
			until
				pad_bytes = 0
			loop
				update_from_byte (0.to_natural_8)
				pad_bytes := pad_bytes - 1
			end
		end

	process_length (length: NATURAL_64)
			-- Update length into the buffer.
			-- (from MESSAGE_DIGEST)
		do
			update_from_byte ((length).to_natural_8)
			update_from_byte ((length |>> 8).to_natural_8)
			update_from_byte ((length |>> 16).to_natural_8)
			update_from_byte ((length |>> 24).to_natural_8)
			update_from_byte ((length |>> 32).to_natural_8)
			update_from_byte ((length |>> 40).to_natural_8)
			update_from_byte ((length |>> 48).to_natural_8)
			update_from_byte ((length |>> 56).to_natural_8)
		end

		end

	reset
			-- Reset computation.
		do
			byte_count := 0.to_natural_64
			buffer_offset := 0
			h1 := 1732584193.to_natural_32
			h2 := 4023233417.to_natural_32
			h3 := 2562383102.to_natural_32
			h4 := 271733878.to_natural_32
		end

	update_from_byte (a_byte: NATURAL_8)
			-- Append byte.
		require -- from  MESSAGE_DIGEST
			True
		local
			l_buffer: like buffer
		do
			l_buffer := buffer
			l_buffer [buffer_offset] := a_byte
			buffer_offset := buffer_offset + 1
			byte_count := byte_count + 1.to_natural_64
			if buffer_offset > l_buffer.upper then
				process_block
				buffer_offset := 0
			end
		ensure then
			buffer_offset_set: buffer_offset = (old buffer_offset + 1) \\ (Block_size)
		end

	update_from_bytes (a_bytes: SPECIAL [NATURAL_8])
			-- Append bytes.
			-- (from MESSAGE_DIGEST)
		local
			i, l_upper: INTEGER_32
		do
			from
				i := 0
				l_upper := a_bytes.upper
			until
				i > l_upper
			loop
				update_from_byte (a_bytes [i])
				i := i + 1
			end
		end

	update_from_io_medium (a_io_medium: IO_MEDIUM)
			-- Append bytes from io medium
			-- (from MESSAGE_DIGEST)
		do
			from
			until
				not a_io_medium.readable
			loop
				a_io_medium.read_natural_8
				if a_io_medium.bytes_read > 0 then
					update_from_byte (a_io_medium.last_natural_8)
				end
			end
		end

	update_from_iterator (a_it: ITERATION_CURSOR [NATURAL_8])

		end

		end

	as_natural_32_le (source: SPECIAL [NATURAL_8]; offset: INTEGER_32): NATURAL_32
			-- NATURAL_32 from bytes.
			-- (from HELPER)
		require -- from HELPER
			valid_start: source.valid_index (offset)
			valid_end: source.valid_index (offset + 3)
		do
			Result := source [offset].to_natural_32
			Result := Result | (source [offset + 1].to_natural_32 |<< 8)
			Result := Result | (source [offset + 2].to_natural_32 |<< 16)
			Result := Result | (source [offset + 3].to_natural_32 |<< 24)
		ensure -- from HELPER
			byte_0: source [offset] = Result.to_natural_8
			byte_1: source [offset + 1] = (Result |>> 8).to_natural_8
			byte_2: source [offset + 2] = (Result |>> 16).to_natural_8
			byte_3: source [offset + 3] = (Result |>> 24).to_natural_8
		end

	ch (u: NATURAL_32; v: NATURAL_32; w: NATURAL_32): NATURAL_32
			-- (from HELPER)
		do
			Result := (u & v) | (u.bit_not & w)
		end

	from_natural_32_be (source: NATURAL_32; target: SPECIAL [NATURAL_8]; offset: INTEGER_32)
			-- (from HELPER)
		require -- from HELPER
			valid_start: target.valid_index (offset)
			valid_end: target.valid_index (offset + 3)
		do
			target [offset] := (source |>> 24).to_natural_8
			target [offset + 1] := (source |>> 16).to_natural_8
			target [offset + 2] := (source |>> 8).to_natural_8
			target [offset + 3] := source.to_natural_8
		ensure -- from HELPER
			byte_0: target [offset] = (source |>> 24).to_natural_8
			byte_1: target [offset + 1] = (source |>> 16).to_natural_8
			byte_2: target [offset + 2] = (source |>> 8).to_natural_8
			byte_3: target [offset + 3] = source.to_natural_8
		end

	from_natural_32_le (source: NATURAL_32; target: SPECIAL [NATURAL_8]; offset: INTEGER_32)
			-- Put `source' into `target'.
			-- (from HELPER)
		require -- from HELPER
			valid_start: target.valid_index (offset)
			valid_end: target.valid_index (offset + 3)
		do
			target [offset] := source.to_natural_8
			target [offset + 1] := (source |>> 8).to_natural_8
			target [offset + 2] := (source |>> 16).to_natural_8
			target [offset + 3] := (source |>> 24).to_natural_8
		ensure -- from HELPER
			byte_0: target [offset] = source.to_natural_8
			byte_1: target [offset + 1] = (source |>> 8).to_natural_8
			byte_2: target [offset + 2] = (source |>> 16).to_natural_8
			byte_3: target [offset + 3] = (source |>> 24).to_natural_8
		end

	maj (u: NATURAL_32; v: NATURAL_32; w: NATURAL_32): NATURAL_32
			-- (from HELPER)
		do
			Result := (u & v) | (u & w) | (v & w)
		end

	parity (u: NATURAL_32; v: NATURAL_32; w: NATURAL_32): NATURAL_32
			-- (from HELPER)
		do
			Result := u.bit_xor (v).bit_xor (w)
		end

	rotate_left_32 (in: NATURAL_32; count: INTEGER_32): NATURAL_32
			-- Left rotate operation
			-- (from HELPER)
		require -- from HELPER
			count_too_small: count >= 0
			count_too_big: count <= 32
		do
			Result := (in |<< count) | (in |>> (32 - count))
		ensure -- from HELPER
			rotate_definition: Result = (in |<< count) | (in |>> (32 - count))
		end

	rotate_right_32 (in: NATURAL_32; count: INTEGER_32): NATURAL_32
			-- (from HELPER)
		require -- from HELPER
			count_too_small: count >= 0
			count_too_big: count <= 32
		do
			Result := (in |>> count) | (in |<< (32 - count))
		ensure -- from HELPER
			rotate_definition: Result = (in |>> count) | (in |<< (32 - count))
		end
	
feature {NONE} -- Implemetation

	buffer: SPECIAL [NATURAL_8]
			-- Buffer

	buffer_offset: INTEGER_32
			-- Buffer offset

	process_block
			-- Process current block in buffer.
		require
			buffer_offset_reset: buffer_offset = 64
		local
			a, b, c, d, f, g, temp: NATURAL_32
			i: INTEGER_32
			l_sines: like Integer_sines
			l_schedule: like schedule
			l_shifts: like Shifts_per_round
		do
			a := h1
			b := h2
			c := h3
			d := h4
			schedule_buffer
			l_sines := Integer_sines
			l_schedule := schedule
			l_shifts := Shifts_per_round
			from
				i := 0
			until
				i > 63
			loop
				if (i < 16) then
					f := (b & c) | (b.bit_not & d)
					g := i.to_natural_32
				elseif (i < 32) then
					f := (d & b) | (d.bit_not & c)
					g := (5.to_natural_32 * i.to_natural_32 + 1.to_natural_32) \\ 16.to_natural_32
				elseif (i < 48) then
					f := b.bit_xor (c).bit_xor (d)
					g := (3.to_natural_32 * i.to_natural_32 + 5.to_natural_32) \\ 16.to_natural_32
				else
					f := c.bit_xor (b | (d.bit_not))
					g := (7.to_natural_32 * i.to_natural_32) \\ 16.to_natural_32
				end
				temp := d
				d := c
				c := b
				b := b + rotate_left_32 ((a + f + l_sines [i + 1] + l_schedule [g.to_integer_32]), l_shifts [i + 1])
				a := temp
				i := i + 1
			end
			h1 := h1 + a
			h2 := h2 + b
			h3 := h3 + c
			h4 := h4 + d
		end

	schedule: SPECIAL [NATURAL_32]
			-- Schedule buffer

	schedule_buffer
			-- Put raw bytes into schedule.
		require
			buffer_offset_reset: buffer_offset = 64
		local
			i, j, l_upper: INTEGER_32
			l_buffer: like buffer
			l_schedule: like schedule
		do
			l_buffer := buffer
			l_schedule := schedule
			from
				j := l_schedule.Lower
				i := l_buffer.Lower
				l_upper := l_buffer.upper
			until
				i > l_upper
			loop
				l_schedule [j] := as_natural_32_le (l_buffer, i)
				i := i + 4
				j := j + 1
			end
		end
