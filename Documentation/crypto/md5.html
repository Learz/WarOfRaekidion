note
	description: "[
		MD5 encryption
		
		Usage: 
			1. Call one or more update* to input source to encode.
			2. Call *digest* to get the computed result. 
			
			Note that calls of *digest* does not reset previous result.
			To start new computation, reset must be called.
	]"
	date: "$Date: 2013-09-11 01:51:12 -0700 (Wed, 11 Sep 2013) $"
	revision: "$Revision: 92930 $"

class 
	MD5

inherit
	MESSAGE_DIGEST

	HELPER

create 
	make

feature {NONE} -- Initialization

	make
			-- Initialize
		do
			create schedule.make_filled (0.to_natural_32, Block_size // 4)
			create buffer.make_filled (0.to_natural_8, Block_size)
			reset
		end
	
feature -- Access

	digest: SPECIAL [NATURAL_8]
			-- Final digest
		local
			l_buffer_offset: like buffer_offset
			l_byte_count: like byte_count
			l_h1, l_h2, l_h3, l_h4: NATURAL_32
		do
			l_buffer_offset := buffer_offset
			l_byte_count := byte_count
			l_h1 := h1
			l_h2 := h2
			l_h3 := h3
			l_h4 := h4
			finish
			create Result.make_filled (0.to_natural_8, Digest_count)
			from_natural_32_le (h1, Result, 0)
			from_natural_32_le (h2, Result, 4)
			from_natural_32_le (h3, Result, 8)
			from_natural_32_le (h4, Result, 12)
			buffer_offset := l_buffer_offset
			byte_count := l_byte_count
			h1 := l_h1
			h2 := l_h2
			h3 := l_h3
			h4 := l_h4
		end

	Digest_count: INTEGER_32 = 16
			-- Length in bytes of the digest

	Block_size: INTEGER_32 = 64
			-- Block size in bytes
	
feature -- Element Change

	reset
			-- Reset computation.
		do
			byte_count := 0.to_natural_64
			buffer_offset := 0
			h1 := 1732584193.to_natural_32
			h2 := 4023233417.to_natural_32
			h3 := 2562383102.to_natural_32
			h4 := 271733878.to_natural_32
		end

	update_from_byte (a_byte: NATURAL_8)
			-- Append byte.
		local
			l_buffer: like buffer
		do
			l_buffer := buffer
			l_buffer [buffer_offset] := a_byte
			buffer_offset := buffer_offset + 1
			byte_count := byte_count + 1.to_natural_64
			if buffer_offset > l_buffer.upper then
				process_block
				buffer_offset := 0
			end
		ensure then
			buffer_offset_set: buffer_offset = (old buffer_offset + 1) \\ (Block_size)
		end
	
feature {NONE} -- Implemetation

	process_block
			-- Process current block in buffer.
		require
			buffer_offset_reset: buffer_offset = 64
		local
			a, b, c, d, f, g, temp: NATURAL_32
			i: INTEGER_32
			l_sines: like Integer_sines
			l_schedule: like schedule
			l_shifts: like Shifts_per_round
		do
			a := h1
			b := h2
			c := h3
			d := h4
			schedule_buffer
			l_sines := Integer_sines
			l_schedule := schedule
			l_shifts := Shifts_per_round
			from
				i := 0
			until
				i > 63
			loop
				if (i < 16) then
					f := (b & c) | (b.bit_not & d)
					g := i.to_natural_32
				elseif (i < 32) then
					f := (d & b) | (d.bit_not & c)
					g := (5.to_natural_32 * i.to_natural_32 + 1.to_natural_32) \\ 16.to_natural_32
				elseif (i < 48) then
					f := b.bit_xor (c).bit_xor (d)
					g := (3.to_natural_32 * i.to_natural_32 + 5.to_natural_32) \\ 16.to_natural_32
				else
					f := c.bit_xor (b | (d.bit_not))
					g := (7.to_natural_32 * i.to_natural_32) \\ 16.to_natural_32
				end
				temp := d
				d := c
				c := b
				b := b + rotate_left_32 ((a + f + l_sines [i + 1] + l_schedule [g.to_integer_32]), l_shifts [i + 1])
				a := temp
				i := i + 1
			end
			h1 := h1 + a
			h2 := h2 + b
			h3 := h3 + c
			h4 := h4 + d
		end

	schedule_buffer
			-- Put raw bytes into schedule.
		require
			buffer_offset_reset: buffer_offset = 64
		local
			i, j, l_upper: INTEGER_32
			l_buffer: like buffer
			l_schedule: like schedule
		do
			l_buffer := buffer
			l_schedule := schedule
			from
				j := l_schedule.Lower
				i := l_buffer.Lower
				l_upper := l_buffer.upper
			until
				i > l_upper
			loop
				l_schedule [j] := as_natural_32_le (l_buffer, i)
				i := i + 4
				j := j + 1
			end
		end

	buffer: SPECIAL [NATURAL_8]
			-- Buffer

	buffer_offset: INTEGER_32
			-- Buffer offset

	schedule: SPECIAL [NATURAL_32]
			-- Schedule buffer
	
feature {NONE} -- Access
