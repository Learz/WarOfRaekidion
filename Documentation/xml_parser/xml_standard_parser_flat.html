feature -- Comparison

	frozen deep_equal (a: detachable ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void
			-- or attached to isomorphic object structures?
			-- (from ANY)
		do
			if a = Void then
				Result := b = Void
			else
				Result := b /= Void and then a.is_deep_equal (b)
			end
		ensure -- from ANY
			shallow_implies_deep: standard_equal (a, b) implies Result
			both_or_none_void: (a = Void) implies (Result = (b = Void))
			same_type: (Result and (a /= Void)) implies (b /= Void and then a.same_type (b))
			symmetric: Result implies deep_equal (b, a)
		end

	frozen equal (a: detachable ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void or attached
			-- to objects considered equal?
			-- (from ANY)
		do
			if a = Void then
				Result := b = Void
			else
				Result := b /= Void and then a.is_equal (b)
			end
		ensure -- from ANY
			definition: Result = (a = Void and b = Void) or else ((a /= Void and b /= Void) and then a.is_equal (b))
		end

	frozen is_deep_equal (other: attached XML_STANDARD_PARSER): BOOLEAN
			-- Are `Current' and `other' attached to isomorphic object structures?
			-- (from ANY)
		require -- from ANY
			other_not_void: other /= Void
		external
			"built_in"
		ensure -- from ANY
			shallow_implies_deep: standard_is_equal (other) implies Result
			same_type: Result implies same_type (other)
			symmetric: Result implies other.is_deep_equal (Current)
		end

	is_equal (other: attached XML_STANDARD_PARSER): BOOLEAN
			-- Is `other' attached to an object considered
			-- equal to current object?
			-- (from ANY)
		require -- from ANY
			other_not_void: other /= Void
		external
			"built_in"
		ensure -- from ANY
			symmetric: Result implies other ~ Current
			consistent: standard_is_equal (other) implies Result
		end

	frozen standard_equal (a: detachable ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void or attached to
			-- field-by-field identical objects of the same type?
			-- Always uses default object comparison criterion.
			-- (from ANY)
		do
			if a = Void then
				Result := b = Void
			else
				Result := b /= Void and then a.standard_is_equal (b)
			end
		ensure -- from ANY
			definition: Result = (a = Void and b = Void) or else ((a /= Void and b /= Void) and then a.standard_is_equal (b))
		end

	frozen standard_is_equal (other: attached XML_STANDARD_PARSER): BOOLEAN
			-- Is `other' attached to an object of the same type
			-- as current object, and field-by-field identical to it?
			-- (from ANY)
		require -- from ANY
			other_not_void: other /= Void
		external
			"built_in"
		ensure -- from ANY
			same_type: Result implies same_type (other)
			symmetric: Result implies other.standard_is_equal (Current)
		end

	is_valid_name_character (c: like last_character): BOOLEAN
			-- Is `c' a valid character for name or tag value?
			-- NameChar ::= NameStartChar | "-" | "." | [0-9] | #xB7 | [#x0300-#x036F] | [#x203F-#x2040]
		local
			n: NATURAL_32
		do
			Result := is_valid_name_start_character (c)
			if not Result then
				if c = '-'.to_character_32 or c = '.'.to_character_32 or ('0'.to_character_32 <= c and c <= '9'.to_character_32) then
					Result := True
				else
					n := c.natural_32_code
					if n = 183.to_natural_32 or (768.to_natural_32 <= n and n <= 879.to_natural_32) or (8255.to_natural_32 <= n and n <= 8256.to_natural_32) then
						Result := True
					end
				end
			end
		end

	is_valid_name_start_character (c: like last_character): BOOLEAN
			-- Is `c' a valid character for name or tag value?
			-- NameStartChar ::= ":" | [A-Z] | "_" | [a-z] | [#xC0-#xD6] | [#xD8-#xF6] | [#xF8-#x2FF]
			--					     | [#x370-#x37D] | [#x37F-#x1FFF] | [#x200C-#x200D] | [#x2070-#x218F]
			--						 | [#x2C00-#x2FEF] | [#x3001-#xD7FF] | [#xF900-#xFDCF] | [#xFDF0-#xFFFD]
			--						 | [#x10000-#xEFFFF]
		local
			n: NATURAL_32
		do
			if ('a'.to_character_32 <= c and c <= 'z'.to_character_32) or ('A'.to_character_32 <= c and c <= 'Z'.to_character_32) or c = ':'.to_character_32 or c = '_'.to_character_32 then
				Result := True
			else
				n := c.natural_32_code
				if (192.to_natural_32 <= n and n <= 214.to_natural_32) or (216.to_natural_32 <= n and n <= 246.to_natural_32) or (248.to_natural_32 <= n and n <= 767.to_natural_32) or (880.to_natural_32 <= n and n <= 893.to_natural_32) or (895.to_natural_32 <= n and n <= 8191.to_natural_32) or (8204.to_natural_32 <= n and n <= 8205.to_natural_32) or (8304.to_natural_32 <= n and n <= 8591.to_natural_32) or (11264.to_natural_32 <= n and n <= 12271.to_natural_32) or (12289.to_natural_32 <= n and n <= 55295.to_natural_32) or (63744.to_natural_32 <= n and n <= 64975.to_natural_32) or (65008.to_natural_32 <= n and n <= 65533.to_natural_32) or (65536.to_natural_32 <= n and n <= 983039.to_natural_32) then
					Result := True
				end
			end
		end

	resolved_entity (s: READABLE_STRING_32): STRING_32
			-- Resolve `s' as an entity
		require
			s_attached: s /= Void
		local
			s32: detachable STRING_32
		do
			inspect s.item (1).as_lower
			when 'a' then 
				if s.count = 3 and then s.item (2).as_lower = 'm'.to_character_32 and then s.item (3).as_lower = 'p'.to_character_32 then
					s32 := {STRING_32}"&"
				elseif s.count = 4 and then s.item (2).as_lower = 'p'.to_character_32 and then s.item (3).as_lower = 'o'.to_character_32 and then s.item (4).as_lower = 's'.to_character_32 then
					s32 := {STRING_32}"'"
				end
			when 'q' then 
				if s.count = 4 and then s.item (2).as_lower = 'u'.to_character_32 and then s.item (3).as_lower = 'o'.to_character_32 and then s.item (4).as_lower = 't'.to_character_32 then
					s32 := {STRING_32}"%""
				end
			when 'l' then 
				if s.count = 2 and then s.item (2).as_lower = 't'.to_character_32 then
					s32 := {STRING_32}"<"
				end
			when 'g' then 
				if s.count = 2 and then s.item (2).as_lower = 't'.to_character_32 then
					s32 := {STRING_32}">"
				end
			else
			end
			if s32 = Void then
				create s32.make (s.count + 2)
				s32.append_character ('&'.to_character_32)
				s32.append_string_general (s)
				s32.append_character (';'.to_character_32)
			end
			Result := s32
		end

	case_insensitive_same_string_8 (a, b: READABLE_STRING_8): BOOLEAN
			-- `a' and `b' are the same string regardless of casing?
		local
			i, n: INTEGER_32
		do
			n := a.count
			if n = b.count then
				from
					i := 1
					Result := True
				until
					not Result or i > n
				loop
					Result := a [i].as_lower = b [i].as_lower
					i := i + 1
				end
			end
		end

	is_blank (s: READABLE_STRING_GENERAL): BOOLEAN
			-- Is string `s' composed only by space/blank characters?
			-- i.e: ' ', '%T', '%N', '%R'
			-- According to XML specification, space is one of
			-- x20 | #x9 | #xD | #xA
			-- i.e. space, tab, CR, LF.			
		local
			i, n: INTEGER_32
		do
			from
				Result := True
				i := 1
				n := s.count
			until
				i > n or not Result
			loop
				inspect s.code (i)
				when 32, 9, 13, 10 then 
				else
					Result := False
				end
				i := i + 1
			end
		end

feature {NONE} -- Implementation: Byte Order Mark

	bom (s: READABLE_STRING_32): detachable STRING_8
			-- Byte Order Mark at the start of `s'.
		local
			i, n: INTEGER_32
			s32: READABLE_STRING_32
		do
			n := s.count
			if n = 0 then
			else
				from
					i := 1
				until
					i > n or not is_bom_character (s [i])
				loop
					i := i + 1
				end
				if i > 1 then
					s32 := s.substring (1, i - 1)
					if s32.is_valid_as_string_8 then
						Result := s32.to_string_8
					end
				end
			end
		end

	is_bom_character (c: CHARACTER_32): BOOLEAN
		do
			if c.is_character_8 and then c.to_character_8.is_printable then
				Result := False
			else
				Result := True
			end
		end

	new_encoded_buffer (a_encoding: READABLE_STRING_GENERAL; a_char_buffer: XML_CHARACTER_8_INPUT_STREAM): detachable like buffer
			-- Create a new `a_encoding' encoded buffer for `a_char_buffer'.
			-- Could be redefined
		require
			a_encoding_not_empty: a_encoding /= Void and then (not a_encoding.is_empty and a_encoding.is_valid_as_string_8)
			a_char_buffer_attached: a_char_buffer /= Void
		do
			if a_encoding.is_case_insensitive_equal ("UTF-8") then
				create {XML_CHARACTER_8_INPUT_STREAM_UTF8_FILTER} Result.make (a_char_buffer)
			elseif a_encoding.is_case_insensitive_equal ("ISO-8859-1") then
				Result := a_char_buffer
			end
		end
	
feature {NONE} -- Implementation: parse

	on_xml_declaration (a_version: READABLE_STRING_32; a_encoding: detachable READABLE_STRING_32; a_standalone: BOOLEAN)
			-- XML declaration.
		require
			a_version_not_void: a_version /= Void
			a_version_not_empty: a_version.count > 0
		local
			ver: detachable READABLE_STRING_32
			enc: detachable READABLE_STRING_32
		do
			if a_version.is_valid_as_string_8 then
				ver := a_version
				if ver.count >= 3 then
					if ver.item (1).is_digit and ver.item (2) = '.'.to_character_32 and ver.substring (3, ver.count).is_integer then
						if ver.item (1) /= '1'.to_character_32 then
							report_error ({STRING_32}"Unsupported xml version info: %"" + a_version + {STRING_32}"%".")
						end
					else
						ver := Void
					end
				else
					ver := Void
				end
			end
			if ver = Void then
				report_error ({STRING_32}"Invalid xml version info: %"" + a_version + {STRING_32}"%".")
			else
				if a_encoding /= Void then
					if a_encoding.is_valid_as_string_8 then
						enc := a_encoding
						if enc.is_empty then
							enc := Void
						end
					else
						report_error ({STRING_32}"Invalid xml encoding: %"" + a_encoding + {STRING_32}"%".")
					end
				end
			end
			if enc = Void then
				set_encoding (Default_encoding)
			else
				set_encoding (enc)
			end
			callbacks.on_xml_declaration (a_version, a_encoding, a_standalone)
		end

	parse
			-- Parse buffer
		local
			c: like last_character
			l_content: STRING_32
			l_in_prolog: BOOLEAN
			l_bom_detection: BOOLEAN
			buf: like buffer
			l_callbacks: like callbacks
		do
			reset
			buf := buffer
			l_callbacks := callbacks
			l_callbacks.on_start
			from
				l_bom_detection := True
				l_in_prolog := True
				create l_content.make_empty
			until
				parsing_stopped or buf.end_of_input
			loop
				c := next_character
				inspect c
				when '<' then 
					if l_in_prolog then
						if l_bom_detection then
							if attached bom (l_content) as l_bom then
								set_encoding_from_bom (l_bom)
								l_content.remove_head (l_bom.count)
							end
							l_bom_detection := False
						end
						if not is_blank (l_content) then
							report_unexpected_content_in_prolog (l_content)
						end
						l_content.wipe_out
					else
						if not l_content.is_empty then
							l_callbacks.on_content (l_content.string)
							l_content.wipe_out
						end
					end
					c := next_character
					inspect c
					when '?' then 
						parse_processing_instruction
					when '/' then 
						parse_end_tag
					when '!' then 
						c := next_character
						inspect c
						when '-' then 
							rewind_character
							parse_comment
						when '[' then 
							rewind_character
							parse_cdata
						when 'D' then 
							rewind_character
							parse_doctype
						else
							report_error ({STRING_32}"syntax error")
						end
					else
						rewind_character
						l_in_prolog := False
						parse_start_tag
					end
				when '&' then 
					l_content.append_string (next_entity)
				else
					l_content.append_character (c)
				end
			end
			if not parsing_stopped then
				if not l_content.is_empty and then not is_blank (l_content) then
					l_callbacks.on_content (l_content.string)
					l_content.wipe_out
				end
				l_callbacks.on_finish
			end
		end

	parse_cdata
			-- Parse for [CDATA[ ..... ]]>
		local
			c: like last_character
			s: STRING_32
			done: BOOLEAN
		do
			c := next_character
			if c /= '['.to_character_32 then
				report_error ({STRING_32}"invalid syntax <!" + c.out.to_string_32)
			else
				c := next_character
				from
					create s.make (6)
				until
					c = '['.to_character_32
				loop
					s.append_character (c.as_upper)
					c := next_character
				end
				if s.same_string_general (Str_cdata) then
					s.wipe_out
					from
						c := next_character
					until
						done or parsing_stopped
					loop
						if c = ']'.to_character_32 then
							c := next_character
							if c = ']'.to_character_32 then
								c := next_character
								if c = '>'.to_character_32 then
									done := True
								elseif c = '-'.to_character_32 then
									s.append_character (']'.to_character_32)
									rewind_character
								else
									s.append_character (']'.to_character_32)
									s.append_character (']'.to_character_32)
									s.append_character (c)
									c := next_character
								end
							else
								s.append_character (']'.to_character_32)
								s.append_character (c)
								c := next_character
							end
						else
							s.append_character (c)
							c := next_character
						end
					end
					if parsing_stopped then
						report_error ({STRING_32}"could not find end of CDATA content")
					else
						callbacks.on_content (s)
					end
				else
					report_error ({STRING_32}"invalid syntax <![" + s)
				end
			end
		end

	parse_comment
			-- Parse for comment
		local
			c: like last_character
			s: like new_string_comment_value
			done: BOOLEAN
		do
			c := next_character
			if c /= '-'.to_character_32 then
				report_error ({STRING_32}"invalid comment syntax")
			else
				c := next_character
				if c /= '-'.to_character_32 then
					report_error ({STRING_32}"invalid comment syntax")
				else
					create s.make_empty
					from
						c := next_character
					until
						done or parsing_stopped
					loop
						if c = '-'.to_character_32 then
							c := next_character
							if c = '-'.to_character_32 then
								c := next_character
								if c = '>'.to_character_32 then
									done := True
								elseif c = '-'.to_character_32 then
									s.append_character ('-'.to_character_32)
									rewind_character
								else
									s.append_character ('-'.to_character_32)
									s.append_character ('-'.to_character_32)
									s.append_character (c)
									c := next_character
								end
							else
								s.append_character ('-'.to_character_32)
								s.append_character (c)
								c := next_character
							end
						else
							s.append_character (c)
							c := next_character
						end
					end
					if parsing_stopped then
						report_error ({STRING_32}"could not find end of comment")
					else
						callbacks.on_comment (s)
					end
				end
			end
		end

	parse_declaration
			-- Parse for xml declaration
		local
			c: like last_character
			l_version, l_encoding: detachable READABLE_STRING_32
			l_standalone: BOOLEAN
			l_name: READABLE_STRING_32
			done: BOOLEAN
			att: like next_attribute_data
		do
			l_standalone := True
			if not parsing_stopped then
				c := current_character
				if c.is_space then
					c := next_non_space_character
				end
				inspect c
				when '?' then 
					c := next_character
					if c = '>'.to_character_32 then
						if l_version /= Void then
							on_xml_declaration (l_version, l_encoding, l_standalone)
						else
							report_error ({STRING_32}"missing version in xml declaration")
						end
					else
						report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' after closing ? in xml declaration")
					end
				else
					rewind_character
					from
						done := False
					until
						done or parsing_stopped
					loop
						att := next_attribute_data
						if att /= Void then
							l_name := att.local_part
							if l_name.is_case_insensitive_equal (Str_version) then
								l_version := att.value
							elseif l_name.is_case_insensitive_equal (Str_encoding) then
								l_encoding := att.value
							elseif l_name.is_case_insensitive_equal (Str_standalone) then
								l_standalone := True
							else
								report_error ({STRING_32}"unknown xml declaration attribute [" + l_name.to_string_32 + {STRING_32}"]")
							end
							rewind_character
						else
							c := current_character
							if c = '?'.to_character_32 then
								c := next_non_space_character
								if c = '>'.to_character_32 then
									done := True
									if l_version /= Void then
										on_xml_declaration (l_version, l_encoding, l_standalone)
									else
										report_error ({STRING_32}"missing version in xml declaration")
									end
								else
									report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' after closing ? in xml declaration")
								end
							end
							done := True
						end
					end
				end
			end
		end

	parse_doctype
			-- Parse for [CDATA[ ..... ]]>
		local
			c: like last_character
			s: STRING_32
			l_depth: INTEGER_32
			in_double_quote: BOOLEAN
			done: BOOLEAN
		do
			c := next_character
			if c /= 'D'.to_character_32 then
				report_error ({STRING_32}"invalid syntax <!" + c.out.to_string_32)
			else
				from
					create s.make (6)
				until
					c.is_space or c = '>'.to_character_32
				loop
					s.append_character (c.as_upper)
					c := next_character
				end
				if s.same_string_general (Str_doctype) then
					create s.make_empty
					from
						c := next_character
					until
						done or parsing_stopped
					loop
						if in_double_quote then
							if c = '"'.to_character_32 then
								in_double_quote := False
							end
							s.append_character (c)
							c := next_character
						else
							inspect c
							when '"' then 
								in_double_quote := True
								s.append_character (c)
								c := next_character
							when '<' then 
								l_depth := l_depth + 1
								s.append_character (c)
								c := next_character
							when '>' then 
								if l_depth = 0 then
									done := True
								else
									l_depth := l_depth - 1
									s.append_character (c)
									c := next_character
								end
							else
								s.append_character (c)
								c := next_character
							end
						end
					end
					if parsing_stopped then
						report_error ({STRING_32}"could not find end of DOCTYPE content")
					else
						register_doctype (s)
					end
				else
					report_error ({STRING_32}"invalid syntax <!" + s)
				end
			end
		end

	parse_end_tag
			-- Parse for end tag
		local
			t: like next_tag
			c: like last_character
		do
			t := next_tag
			c := current_character
			if c.is_space then
				c := next_non_space_character
			end
			if c = '>'.to_character_32 then
				callbacks.on_end_tag (Void, t.prefix_part, t.local_part)
			else
				report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' in end tag")
			end
		end

	parse_processing_instruction
			-- Parse for  <?pi .... ?>
		local
			c: like last_character
			n, s: STRING_32
			done: BOOLEAN
		do
			c := next_character
			from
				create n.make (6)
			until
				c.is_space or c = '?'.to_character_32
			loop
				n.append_character (c)
				c := next_character
			end
			if n.is_empty then
				rewind_character
				report_error ({STRING_32}"Invalid processing instruction syntax <?" + c.out.to_string_32)
			elseif n.is_case_insensitive_equal (Str_xml) then
				parse_declaration
			else
				create s.make_empty
				from
					c := next_character
				until
					done or parsing_stopped
				loop
					if c = '?'.to_character_32 then
						c := next_character
						if c = '>'.to_character_32 then
							done := True
						elseif c = '?'.to_character_32 then
							s.append_character ('?'.to_character_32)
						else
							s.append_character ('?'.to_character_32)
							s.append_character (c)
							c := next_character
						end
					else
						s.append_character (c)
						c := next_character
					end
				end
				if parsing_stopped then
					report_error ({STRING_32}"could not find end of Processing Instruction")
				else
					callbacks.on_processing_instruction (n, s)
				end
			end
		end

	parse_start_tag
			-- Parse for start tag.
		local
			t: like next_tag
			c: like last_character
			att: like next_attribute_data
			done: BOOLEAN
			l_callbacks: like callbacks
		do
			t := next_tag
			l_callbacks := callbacks
			l_callbacks.on_start_tag (Void, t.prefix_part, t.local_part)
			c := current_character
			if c.is_space then
				c := next_non_space_character
			end
			inspect c
			when '>' then 
				l_callbacks.on_start_tag_finish
			when '/' then 
				c := next_character
				if c = '>'.to_character_32 then
					l_callbacks.on_start_tag_finish
					l_callbacks.on_end_tag (Void, t.prefix_part, t.local_part)
				else
					report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' after closing / in start tag")
				end
			else
				rewind_character
				from
					done := False
				until
					done or parsing_stopped
				loop
					att := next_attribute_data
					if att /= Void then
						l_callbacks.on_attribute (Void, att.prefix_part, att.local_part, att.value)
						rewind_character
					else
						c := current_character
						inspect c
						when '>' then 
							l_callbacks.on_start_tag_finish
						when '/' then 
							c := next_character
							if c = '>'.to_character_32 then
								done := True
								l_callbacks.on_start_tag_finish
								l_callbacks.on_end_tag (Void, t.prefix_part, t.local_part)
							else
								report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' after closing / in start tag")
							end
						else
							report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' in start tag")
						end
						done := True
					end
				end
			end
		end

	report_error (a_message: READABLE_STRING_32)
			-- Report error with message `a_message'
		require
			a_message_attached: a_message /= Void
		local
			s: STRING_32
			p: like position
		do
			p := position
			unset_checkpoint_position
			create s.make (a_message.count)
			s.append_string_general (a_message)
			create error_message.make_from_string (s)
			if error_position = Void then
				error_position := p
			end
			parsing_stopped := True
			s.append_character (' '.to_character_32)
			s.append_character ('('.to_character_32)
			s.append_string ({STRING_32}"position=")
			s.append_string (p.to_string_32)
			callbacks.on_error (s)
		ensure
			error_occurred: error_occurred
		end

	report_missing_attribute_value (p: detachable READABLE_STRING_32; n: READABLE_STRING_32)
			-- Report missing value for attribute `p:n'
		do
			report_error ({STRING_32}"Attributes without any value are forbidden")
		end

	report_unexpected_content (a_content: STRING_8)
			-- Report unexpected content `a_content'
		do
			report_error ({STRING_32}"Unexpected content (not well-formed XML)")
		end

	report_unexpected_content_in_prolog (a_content: READABLE_STRING_32)
			-- Report unexpected content `a_content' in prolog
		do
			report_error ({STRING_32}"Unexpected content in prolog (not well-formed XML)")
		end

	report_unsupported_bom_value (a_bom: READABLE_STRING_GENERAL)
			-- Report unsupported BOM value for `a_bom'
		do
			report_error ({STRING_32}"Unsupported BOM value %"" + a_bom.to_string_32 + {STRING_32}"%".")
		end

	report_warning (a_message: READABLE_STRING_32)
			-- Report warning with message `a_message'
		require
			a_message_attached: a_message /= Void
		local
			s: STRING_32
			p: like position
		do
			p := position
			create s.make (a_message.count)
			s.append_string_general (a_message)
			warning_message := s
		end

feature -- Parsing

	parse_from_file (a_file: FILE)
			-- Parse from file `a_file'
		require -- from XML_PARSER
			a_file_attached: a_file /= Void
			a_file_is_readable: a_file.is_open_read and then a_file.is_readable
		local
			s: XML_FILE_INPUT_STREAM
		do
			create s.make (a_file)
			s.compute_smart_chunk_size (a_file.count, 16384)
			s.start
			parse_from_stream (s)
			s.close
		end

	parse_from_filename (a_filename: READABLE_STRING_GENERAL)
		obsolete "Use parse_from_path [2012-oct]"
			-- Parse from file named `a_filename'
			-- (from XML_PARSER)
		require -- from XML_PARSER
			a_filename_valid: a_filename /= Void and then not a_filename.is_empty
		local
			p: PATH
		do
			create p.make_from_string (a_filename)
			parse_from_path (p)
		end

	parse_from_path (a_path: PATH)
			-- Parse from file named `a_path'
		require -- from XML_PARSER
			a_path_valid: a_path /= Void and then not a_path.is_empty
		local
			f: detachable RAW_FILE
			retried: BOOLEAN
		do
			if not retried then
				create f.make_with_path (a_path)
				if f.exists and f.is_readable then
					f.open_read
					parse_from_file (f)
					f.close
				else
					report_error ({STRING_32}"Unable to open file %"" + a_path.name + {STRING_32}"%".")
				end
			else
				report_error ({STRING_32}"Error when trying to open file %"" + a_path.name + {STRING_32}"%".")
				if f /= Void and then not f.is_closed then
					f.close
				end
			end
		rescue
			if not retried then
				retried := True
				retry
			end
		end

	parse_from_stream (a_stream: XML_INPUT_STREAM)
			-- Parse stream `a_stream'
		require -- from XML_PARSER
			a_stream_attached: a_stream /= Void
		do
			set_buffer (a_stream)
			parse
			set_buffer (Empty_buffer)
		end

	parse_from_string (a_string: READABLE_STRING_GENERAL)
			-- Parse string general `a_string'
			-- (from XML_PARSER)
		require -- from XML_PARSER
			a_string_attached: a_string /= Void
		do
			if attached {READABLE_STRING_8} a_string as s8 then
				parse_from_string_8 (s8)
			elseif attached {READABLE_STRING_32} a_string as s32 then
				parse_from_string_32 (s32)
			else
				check
					is_string_32: a_string.is_string_32
				end
				parse_from_string_32 (a_string.to_string_32)
			end
		end

	parse_from_string_32 (a_string: READABLE_STRING_32)
			-- Parse string `a_string'
		require -- from XML_PARSER
			a_string_attached: a_string /= Void
		local
			s: XML_STRING_32_INPUT_STREAM
		do
			create s.make (a_string)
			parse_from_stream (s)
			s.close
		end

	parse_from_string_8 (a_string: READABLE_STRING_8)
			-- Parse string `a_string'
		require -- from XML_PARSER
			a_string_attached: a_string /= Void
		local
			s: XML_STRING_INPUT_STREAM
		do
			create s.make (a_string)
			parse_from_stream (s)
			s.close
		end

	internal_read_character (buf: like buffer): like last_character
			-- Internal implementation of next_character
		require
			buf_attached: buf /= Void
			buf_not_end_of_input: not buf.end_of_input
		local
			c: NATURAL_32
			cr_code: NATURAL_32
		do
			buf.read_character_code
			cr_code := ('%R').natural_32_code
			c := buf.last_character_code
			if c = cr_code then
				from
				until
					c /= cr_code or buf.end_of_input
				loop
					buf.read_character_code
					c := buf.last_character_code
				end
			end
			Result := c.to_character_32
		end

	last_character: CHARACTER_32
			-- Last read character

	next_attribute_data: detachable TUPLE [prefix_part: detachable READABLE_STRING_32; local_part: READABLE_STRING_32; value: READABLE_STRING_32]
			-- Return next attribute's data (prefix,local and value)
			-- move index
		local
			c: like last_character
			n: READABLE_STRING_32
			p, v: detachable READABLE_STRING_32
			l_was_space: BOOLEAN
			att_name: like next_attribute_name
		do
			att_name := next_attribute_name
			n := att_name.local_part
			p := att_name.prefix_part
			if not n.is_empty then
				c := current_character
				set_checkpoint_position
				if c.is_space then
					l_was_space := True
					c := next_non_space_character
				end
				if c = '='.to_character_32 then
					unset_checkpoint_position
				elseif l_was_space or c = '>'.to_character_32 or c = '/'.to_character_32 then
					report_missing_attribute_value (p, n)
					Result := [p, n, {STRING_32}""]
				else
					report_error ({STRING_32}"unexpected character '" + character_output (c) + {STRING_32}"' in attribute name")
				end
				if Result = Void and not parsing_stopped then
					check
						c_is_equal_sign: c = '='.to_character_32
					end
					set_checkpoint_position
					c := next_character
					if c.is_space then
						c := next_non_space_character
					end
					rewind_character
					v := next_attribute_value
					unset_checkpoint_position
					Result := [p, n, v]
					c := current_character
					if c.is_space or c = '>'.to_character_32 or c = '/'.to_character_32 or c = '?'.to_character_32 then
					else
						report_error ({STRING_32}"unexpected character '" + character_output (current_character) + {STRING_32}"' after attribute declaration (expecting white space or '>' or '/' or '?' )")
					end
				end
			end
		ensure
			current_character.is_space or current_character = '>'.to_character_32 or current_character = '/'.to_character_32 or current_character = '?'.to_character_32
		end

	next_attribute_name: TUPLE [prefix_part: detachable READABLE_STRING_32; local_part: READABLE_STRING_32]
			-- Return next attribute prefix and local part.
			-- move index
		local
			c: like last_character
			n: like new_string_attribute_name
			p: INTEGER_32
		do
			n := new_string_attribute_name
			c := next_non_space_character
			if is_valid_name_start_character (c) then
				from
					n.append_character (c)
					c := next_character
				until
					not is_valid_name_character (c)
				loop
					n.append_character (c)
					c := next_character
				end
			end
			p := n.index_of (':'.to_character_32, 1)
			if p > 1 then
				Result := [n.substring (1, p - 1), n.substring (p + 1, n.count)]
			else
				Result := [Void, n.string]
			end
			n.wipe_out
		ensure
			new_string_attribute_name_is_empty: new_string_attribute_name.is_empty
		end

	next_attribute_value: READABLE_STRING_32
			-- Return next attribute's value
			-- move index
		local
			c: like last_character
			l_in_double_quote: BOOLEAN
			l_in_single_quote: BOOLEAN
			done: BOOLEAN
			s: like new_string_attribute_value
		do
			s := new_string_attribute_value
			c := next_character
			if c = '"'.to_character_32 then
				l_in_double_quote := True
				c := next_character
			elseif c = '%''.to_character_32 then
				l_in_single_quote := True
				c := next_character
			elseif c.is_space or c = '/'.to_character_32 or c = '>'.to_character_32 then
				report_error ({STRING_32}"unexpected space after = in attribute declaration")
			end
			if not parsing_stopped then
				from
					done := False
				until
					done or parsing_stopped
				loop
					if not l_in_double_quote and not l_in_single_quote and (c.is_space or c = {CHARACTER_32}'/' or c = {CHARACTER_32}'>') then
						done := True
					else
						inspect c
						when '&' then 
							s.append_string (next_entity)
						when '"' then 
							if l_in_double_quote then
								done := True
							elseif l_in_single_quote then
								s.append_character (c)
							else
								report_error ({STRING_32}"unexpected %" in attribute value")
							end
						when '%'' then 
							if l_in_single_quote then
								done := True
							elseif l_in_double_quote then
								s.append_character (c)
							else
								report_error ({STRING_32}"unexpected ' in attribute value")
							end
						else
							s.append_character (c)
						end
						c := next_character
					end
				end
			end
			if parsing_stopped then
				create {STRING_32} Result.make_empty
			else
				Result := s.string
			end
			s.wipe_out
		ensure
			new_string_attribute_value_is_empty: new_string_attribute_value.is_empty
		end

	next_character: like last_character
			-- Return next character
			-- move index
		local
			buf: like buffer
		do
			Result := rewinded_character
			if Result = '%U'.to_character_32 then
				buf := buffer
				if not buf.end_of_input then
					Result := internal_read_character (buf)
				else
					report_error ({STRING_32}"no more characters")
				end
			else
				rewinded_character := '%U'.to_character_32
			end
			last_character := Result
		end

	next_entity: STRING_32
			-- Return next entity value
			-- move index
		local
			c: like last_character
			is_char: BOOLEAN
			is_hexa: BOOLEAN
			s: STRING_32
		do
			create s.make_empty
			c := next_character
			if c = '#'.to_character_32 then
				is_char := True
				c := next_character
				if c = 'x'.to_character_32 then
					is_hexa := True
					c := next_character
				end
			end
			if is_char then
				if is_hexa then
					from
					until
						not c.is_hexa_digit or c = ';'.to_character_32
					loop
						s.append_character (c)
						c := next_character
					end
				else
					from
					until
						not c.is_digit or c = ';'.to_character_32
					loop
						s.append_character (c)
						c := next_character
					end
				end
			else
				from
				until
					not is_valid_entity_character (c) or c = ';'.to_character_32
				loop
					s.append_character (c)
					c := next_character
				end
			end
			if c = ';'.to_character_32 then
				if is_char then
					if is_hexa then
						create Result.make (1)
						Result.append_code (hexa_string_to_natural_32 (s))
					elseif s.is_integer then
						create Result.make (1)
						Result.append_code (s.to_natural_32)
					else
						create Result.make (s.count + 2)
						Result.append_character ('&'.to_character_32)
						Result.append_string (s)
						Result.append_character (';'.to_character_32)
					end
				else
					Result := resolved_entity (s)
				end
			else
				rewind_character
				create Result.make (s.count + 2)
				Result.append_character ('&'.to_character_32)
				Result.append_string (s)
				report_error ({STRING_32}"Invalid entity syntax " + s)
			end
		end

	next_non_space_character: like last_character
			-- Return next character which is not a space
			-- move index
		do
			from
				Result := next_character
			until
				not Result.is_space
			loop
				Result := next_character
			end
		ensure
			not_a_space: not Result.is_space
		end

	next_tag: TUPLE [prefix_part: detachable STRING_32; local_part: STRING_32]
			-- Return next tag value
			-- move index
		local
			c: like last_character
			p: detachable like new_string_tag
			s: like new_string_tag
		do
			s := new_string_tag
			c := next_non_space_character
			if is_valid_tag_start_character (c) then
				from
				until
					not is_valid_tag_character (c)
				loop
					if c = ':'.to_character_32 and p = Void then
						p := s.string
						s.wipe_out
					else
						s.append_character (c)
					end
					c := next_character
				end
			end
			Result := [p, s.string]
			s.wipe_out
		ensure
			new_string_tag_is_empty: new_string_tag.is_empty
		end

	rewind_character
			-- Rewind by 1 character
		require
			rewinded_character_null: rewinded_character = '%U'.to_character_32
		do
			rewinded_character := last_character
		ensure
			rewinded_character_valid: rewinded_character /= '%R'.to_character_32
		end

	rewinded_character: like last_character
			-- Last character put in stack
	
feature {NONE} -- Retrieval

	frozen internal_correct_mismatch
			-- Called from runtime to perform a proper dynamic dispatch on `correct_mismatch'
			-- from MISMATCH_CORRECTOR.
			-- (from ANY)
		local
			l_msg: STRING_8
			l_exc: EXCEPTIONS
		do
			if attached {MISMATCH_CORRECTOR} Current as l_corrector then
				l_corrector.correct_mismatch
			else
				create l_msg.make_from_string ("Mismatch: ")
				create l_exc
				l_msg.append (generating_type.name)
				l_exc.raise_retrieval_exception (l_msg)
			end
		end
	
feature -- Status

	is_correct: BOOLEAN
			-- Has no error been detected?
			-- (from XML_PARSER)
		do
			Result := not error_occurred
		end

	last_error: INTEGER_32
			-- Last error's code
			-- See XML_ERROR_CODES.
			-- (from XML_PARSER)
		do
			if error_occurred then
				Result := {XML_ERROR_CODES}.xml_err_unknown
			else
				Result := {XML_ERROR_CODES}.xml_err_none
			end
		end
	
feature -- Status		

	buffer_position: XML_POSITION
			-- XML position in buffer
		do
			create Result.make (buffer.name.as_string_32, byte_index, column, line)
		end

	byte_index: INTEGER_32
			-- Byte index
		do
			Result := buffer.index
		end

	column: INTEGER_32
			-- Last column number
		do
			Result := buffer.column
		end

	error_message: detachable STRING_32
			-- Error message

	error_position: detachable XML_POSITION
			-- Position when error occurred

	line: INTEGER_32
			-- Last line number
		do
			Result := buffer.line
		end

	position: XML_POSITION
			-- Logical position
		do
			if attached checkpoint_position as p then
				Result := p
			else
				Result := buffer_position
			end
		ensure -- from XML_PARSER
			result_attached: Result /= Void
		end

	warning_message: detachable STRING_32
			-- Warning message
	
invariant

	reflexive_equality: standard_is_equal (Current)
	reflexive_conformance: conforms_to (Current)
